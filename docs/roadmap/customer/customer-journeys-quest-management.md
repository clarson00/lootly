### Key Principle: No Explicit Enrollment

Users don't "join" or "enroll" at businesses. They:
1. **Browse** - See all opportunities freely
2. **Add to Quests** - Express intent to pursue
3. **Take Action** - First check-in creates business relationship

The phone number is their identity. Everything else is automatic.

## Database Schema

```sql
-- Customer quest tracking
CREATE TABLE customer_quests (
  id TEXT PRIMARY KEY,                 -- "cquest_abc123"
  customer_id TEXT NOT NULL REFERENCES customers(id),

  -- What they're chasing
  target_type TEXT NOT NULL,           -- 'reward', 'ruleset', 'rule'
  target_id TEXT NOT NULL,             -- reward_id, ruleset_id, rule_id
  business_id TEXT NOT NULL REFERENCES businesses(id),

  -- Status
  status TEXT NOT NULL DEFAULT 'active',  -- 'upcoming', 'active', 'completed', 'removed', 'expired'
  added_at TIMESTAMPTZ DEFAULT NOW(),
  starts_at TIMESTAMPTZ,               -- For scheduled quests
  completed_at TIMESTAMPTZ,
  removed_at TIMESTAMPTZ,

  -- Notifications
  notify_nearby BOOLEAN DEFAULT true,
  notify_progress BOOLEAN DEFAULT true,
  notify_bonus_events BOOLEAN DEFAULT false,
  notify_on_start BOOLEAN DEFAULT true,  -- For upcoming quests

  -- Reminders
  reminder_enabled BOOLEAN DEFAULT false,
  reminder_frequency TEXT,              -- 'daily', 'weekly', 'once'
  reminder_day TEXT,                    -- 'monday', 'friday', etc.
  reminder_time TEXT,                   -- '18:00'
  next_reminder_at TIMESTAMPTZ,

  -- Calendar integration
  calendar_event_id TEXT,               -- If added to device calendar

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),

  UNIQUE(customer_id, target_type, target_id)
);

-- Quest activity log (for analytics)
CREATE TABLE customer_quest_events (
  id TEXT PRIMARY KEY,
  quest_id TEXT NOT NULL REFERENCES customer_quests(id),
  event_type TEXT NOT NULL,             -- 'added', 'started', 'progress', 'completed', 'removed'
  event_data JSONB,                     -- Progress snapshots, etc.
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_quests_customer ON customer_quests(customer_id, status);
CREATE INDEX idx_quests_business ON customer_quests(business_id, status);
CREATE INDEX idx_quests_upcoming ON customer_quests(starts_at)
  WHERE status = 'upcoming';
CREATE INDEX idx_quests_reminders ON customer_quests(next_reminder_at)
  WHERE reminder_enabled = true;
```
